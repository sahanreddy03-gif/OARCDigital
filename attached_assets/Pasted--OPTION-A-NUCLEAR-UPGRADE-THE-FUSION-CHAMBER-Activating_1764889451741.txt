class OrbitingCard {
  constructor(index, total) {
    this.angle = (index / total) * Math.PI * 2;
    this.radius = 600; // Distance from portal
    this.depth = Math.sin(this.angle) * 200; // Z-depth
    this.speed = 0.0005; // Rotation speed
  }
  
  update() {
    this.angle += this.speed;
    
    // Calculate 3D position
    this.x = Math.cos(this.angle) * this.radius;
    this.y = Math.sin(this.angle) * this.radius;
    this.z = Math.sin(this.angle * 2) * this.depth;
    
    // Scale based on Z (perspective)
    this.scale = 1 + (this.z / 1000);
    
    // Blur based on Z
    this.blur = Math.abs(this.z) / 100;
  }
  
  render() {
    element.style.transform = `
      translate3d(${this.x}px, ${this.y}px, ${this.z}px)
      scale(${this.scale})
      rotateY(${-this.angle}rad)
    `;
    element.style.filter = `blur(${this.blur}px)`;
  }
}

// 8 cards orbiting
const cards = [];
for (let i = 0; i < 8; i++) {
  cards.push(new OrbitingCard(i, 8));
}

// Animation loop
function animate() {
  cards.forEach(card => {
    card.update();
    card.render();
  });
  requestAnimationFrame(animate);
}

class QuantumParticle {
  constructor(type, position) {
    this.type = type; // 'creative', 'ai', or 'fusion'
    this.position = position.clone();
    this.velocity = new THREE.Vector3();
    this.acceleration = new THREE.Vector3();
    this.life = 1.0;
    this.trail = [];
  }
  
  applyForce(force) {
    this.acceleration.add(force);
  }
  
  update() {
    // Apply forces
    const portalAttraction = this.getPortalForce();
    const cursorRepulsion = this.getCursorForce();
    
    this.applyForce(portalAttraction);
    this.applyForce(cursorRepulsion);
    
    // Update physics
    this.velocity.add(this.acceleration);
    this.velocity.multiplyScalar(0.98); // Drag
    this.position.add(this.velocity);
    this.acceleration.multiplyScalar(0);
    
    // Update trail
    this.trail.push(this.position.clone());
    if (this.trail.length > 20) this.trail.shift();
    
    // Decay life
    this.life -= 0.01;
    
    // Check if reached portal
    if (this.position.distanceTo(portalCenter) < 50) {
      this.reachedPortal();
    }
  }
  
  getPortalForce() {
    // Gravitational attraction to portal
    const direction = portalCenter.clone().sub(this.position);
    const distance = direction.length();
    const strength = 100 / (distance * distance);
    return direction.normalize().multiplyScalar(strength);
  }
  
  getCursorForce() {
    // Magnetic field from cursor
    const direction = this.position.clone().sub(cursorPosition);
    const distance = direction.length();
    if (distance < 200) {
      const strength = (200 - distance) / 200 * 5;
      return direction.normalize().multiplyScalar(strength);
    }
    return new THREE.Vector3();
  }
  
  reachedPortal() {
    // Explode into fusion particles
    for (let i = 0; i < 5; i++) {
      const angle = Math.random() * Math.PI * 2;
      const speed = 2 + Math.random() * 3;
      const fusion = new QuantumParticle('fusion', this.position);
      fusion.velocity.set(
        Math.cos(angle) * speed,
        Math.sin(angle) * speed,
        (Math.random() - 0.5) * speed
      );
      particles.push(fusion);
    }
    this.life = 0; // Remove original
  }
  
  render(ctx) {
    // Render with bloom effect
    ctx.shadowBlur = 30;
    ctx.shadowColor = this.getColor();
    
    // Draw trail
    ctx.beginPath();
    this.trail.forEach((pos, i) => {
      const alpha = (i / this.trail.length) * this.life;
      ctx.globalAlpha = alpha;
      if (i === 0) {
        ctx.moveTo(pos.x, pos.y);
      } else {
        ctx.lineTo(pos.x, pos.y);
      }
    });
    ctx.strokeStyle = this.getColor();
    ctx.lineWidth = 2;
    ctx.stroke();
    
    // Draw particle
    ctx.globalAlpha = this.life;
    ctx.fillStyle = this.getColor();
    ctx.beginPath();
    ctx.arc(this.position.x, this.position.y, 3, 0, Math.PI * 2);
    ctx.fill();
  }
  
  getColor() {
    switch(this.type) {
      case 'creative': return '#FF6B9D';
      case 'ai': return '#00FF41';
      case 'fusion': return '#FFFFFF';
    }
  }
}

// Spawn particles continuously
setInterval(() => {
  // From creative pieces
  workPieces.forEach(piece => {

  // From AI nodes

}, 50);

class CustomCursor {
  constructor() {
    this.position = { x: 0, y: 0 };
    this.velocity = { x: 0, y: 0 };
    this.size = 20;
    this.targetSize = 20;
    this.trail = [];
  }
  
  update(mouseX, mouseY) {
    // Calculate velocity
    this.velocity.x = mouseX - this.position.x;
    this.velocity.y = mouseY - this.position.y;
    
    // Update position
    this.position.x = mouseX;
    this.position.y = mouseY;
    
    // Size based on speed
    const speed = Math.sqrt(
      this.velocity.x ** 2 + this.velocity.y ** 2
    );
    this.targetSize = 20 + Math.min(speed * 0.5, 30);
    this.size += (this.targetSize - this.size) * 0.2;
    
    // Update trail
    this.trail.push({ ...this.position });
    if (this.trail.length > 10) this.trail.shift();
    
    // Apply magnetic force to nearby elements
    this.applyMagneticField();
  }
  
  applyMagneticField() {
    const radius = 200;
    
    // Affect particles
    particles.forEach(p => {
      const dx = p.position.x - this.position.x;
      const dy = p.position.y - this.position.y;
      const distance = Math.sqrt(dx*dx + dy*dy);
      
      if (distance < radius) {
        const force = (radius - distance) / radius;
        p.velocity.x += dx * force * 0.1;
        p.velocity.y += dy * force * 0.1;
      }
    });
    
    // Affect work cards
    workCards.forEach(card => {
      const distance = getDistance(card.position, this.position);
      if (distance < radius) {
        card.tiltToward(this.position);
      }
    });
  }
  
  render(ctx) {
    // Draw trail
    this.trail.forEach((pos, i) => {
      const alpha = i / this.trail.length * 0.5;
      const size = this.size * (i / this.trail.length);
      
      ctx.globalAlpha = alpha;
      ctx.fillStyle = '#FF6B35';
      ctx.shadowBlur = 20;
      ctx.shadowColor = '#FF6B35';
      ctx.beginPath();
      ctx.arc(pos.x, pos.y, size, 0, Math.PI * 2);
      ctx.fill();
    });
    
    // Draw main cursor
    ctx.globalAlpha = 1;
    ctx.fillStyle = 'rgba(255,107,53,0.3)';
    ctx.strokeStyle = '#FF6B35';
    ctx.lineWidth = 2;
    ctx.shadowBlur = 30;
    ctx.beginPath();
    ctx.arc(this.position.x, this.position.y, this.size, 0, Math.PI * 2);
    ctx.fill();
    ctx.stroke();
  }
  
  click() {
    // Shockwave effect
    const shockwave = {
      x: this.position.x,
      y: this.position.y,
      radius: 0,
      maxRadius: 300,
      strength: 10
    };
    
    // Animate shockwave
    const animate = () => {
      shockwave.radius += 10;
      
      // Push particles
      particles.forEach(p => {
        const distance = getDistance(p.position, shockwave);
        if (Math.abs(distance - shockwave.radius) < 20) {
          const angle = Math.atan2(
            p.position.y - shockwave.y,
            p.position.x - shockwave.x
          );
          p.velocity.x += Math.cos(angle) * shockwave.strength;
          p.velocity.y += Math.sin(angle) * shockwave.strength;
        }
      });
      
      if (shockwave.radius < shockwave.maxRadius) {
        requestAnimationFrame(animate);
      }
    };
    animate();
  }
}

// Initialize

  cursor.update(e.clientX, e.clientY);
});

function playParticleSound(position) {

  const panner = audioCtx.createPanner();
